-- Create new admins table structure
-- Create the database
CREATE DATABASE IF NOT EXISTS cbt_app;
USE cbt_app;

-- 1. active_exams
CREATE TABLE active_exams (
    id INT AUTO_INCREMENT PRIMARY KEY,
    subject VARCHAR(100),
    is_active BOOLEAN,
    exam_date DATETIME
);

-- 2. activities_log
CREATE TABLE activities_log (
    id INT AUTO_INCREMENT PRIMARY KEY,
    activity TEXT,
    admin_id INT,
    ip_address VARCHAR(45),
    user_agent TEXT,
    created_at DATETIME
);

-- 3. admins
CREATE TABLE admins (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(100) UNIQUE,
    password VARCHAR(255),
    created_at DATETIME,
    role VARCHAR(50)
);

-- 4. fill_blank_questions
CREATE TABLE fill_blank_questions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    question_id INT,
    correct_answer TEXT,
    FOREIGN KEY (question_id) REFERENCES new_questions(id) ON DELETE CASCADE
);

-- 5. image_questions
CREATE TABLE image_questions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    question_id INT,
    image_path TEXT,
    description TEXT,
    correct_answer TEXT,
    FOREIGN KEY (question_id) REFERENCES new_questions(id) ON DELETE CASCADE
);

-- 6. multiple_choice_questions
CREATE TABLE multiple_choice_questions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    question_id INT,
    option1 TEXT,
    option2 TEXT,
    option3 TEXT,
    option4 TEXT,
    correct_answers TEXT,
    image_path TEXT,
    FOREIGN KEY (question_id) REFERENCES new_questions(id) ON DELETE CASCADE
);

-- 7. new_questions
CREATE TABLE new_questions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    question_text TEXT,
    test_id INT,
    class VARCHAR(20),
    subject VARCHAR(100),
    question_type VARCHAR(50),
    created_at DATETIME,
    FOREIGN KEY (test_id) REFERENCES tests(id) ON DELETE CASCADE
);

-- 8. results
CREATE TABLE results (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    test_id INT,
    score INT,
    total_questions INT,
    created_at DATETIME,
    status VARCHAR(50),
    reattempt_approved BOOLEAN,
    FOREIGN KEY (test_id) REFERENCES tests(id) ON DELETE CASCADE
);

-- 9. settings
CREATE TABLE settings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    setting_name VARCHAR(100),
    setting_value TEXT
);

-- 10. single_choice_questions
CREATE TABLE single_choice_questions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    question_id INT,
    option1 TEXT,
    option2 TEXT,
    option3 TEXT,
    option4 TEXT,
    correct_answer TEXT,
    image_path TEXT,
    FOREIGN KEY (question_id) REFERENCES new_questions(id) ON DELETE CASCADE
);

-- 11. students
CREATE TABLE students (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(150),
    email VARCHAR(100),
    class VARCHAR(20),
    created_at DATETIME,
    role VARCHAR(50)
);

-- 12. teachers
CREATE TABLE teachers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(100),
    last_name VARCHAR(100),
    username VARCHAR(100) UNIQUE,
    email VARCHAR(100),
    password VARCHAR(255),
    phone VARCHAR(20),
    created_at DATETIME,
    role VARCHAR(50)
);

-- 13. teacher_subjects
CREATE TABLE teacher_subjects (
    id INT AUTO_INCREMENT PRIMARY KEY,
    teacher_id INT,
    subject VARCHAR(100),
    FOREIGN KEY (teacher_id) REFERENCES teachers(id)
);

-- 14. tests
CREATE TABLE tests (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200),
    class VARCHAR(20),
    subject VARCHAR(100),
    created_at DATETIME,
    duration INT
);

-- 15. true_false_questions
CREATE TABLE true_false_questions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    question_id INT,
    correct_answer BOOLEAN,
    FOREIGN KEY (question_id) REFERENCES new_questions(id) ON DELETE CASCADE
);
